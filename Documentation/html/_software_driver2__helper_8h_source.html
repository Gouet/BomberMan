288 REALINLINE u32 PixelAdd32 ( const u32 c2, const u32 c1)
289 {
290  u32 sum = ( c2 & 0x00FFFFFF ) + ( c1 & 0x00FFFFFF );
291  u32 low_bits = ( c2 ^ c1 ) & 0x00010101;
292  s32 carries = ( sum - low_bits ) & 0x01010100;
